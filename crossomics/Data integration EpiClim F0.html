Summary

# Build the transcriptomic table
transcripto_sig <- do.call(rbind, lapply(names(kegg_list), function(cond) {
  g <- kegg_list[[cond]]
  if (is.null(g) || nrow(g) == 0) return(NULL)
  df <- as.data.frame(g)
  df$Condition <- cond
  df[, c("Description", "Condition")]
})) %>%
  rename(Pathway = Description) %>%
  mutate(omics = "Transcriptomics")

# Filtrer "Global and overview maps" comme dans Figure 3C
if (exists("kegg_combined")) {
  transcripto_sig <- transcripto_sig %>%
    left_join(kegg_combined %>% 
                select(Description, category_B) %>% 
                distinct(),
              by = c("Pathway" = "Description")) %>%
    filter(is.na(category_B) | category_B != "Global and overview maps") %>%
    select(-category_B)
}
